def initialize(path)
  # open
  @db = Sophia.new path

  # close the db on exit to prevent corruption
  at_exit do
    @db.close
    exit
  end
end

def value?(value)
  # @db.include?(value)
  @db.each_value do |v|
    return true if v == value
  end

  false
end
